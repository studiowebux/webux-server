"use strict";

/**
 * try another port to start the server.
 * @author Tommy Gingras <tommy@studiowebux.com>
 * @module webux-server/change-port
 * @param {number} port The actual port, Mandatory
 * @returns {Number} The new port
 * @throws will throw an error if the number is not between 0 and 65535
 */
const changePort = port => {
  if (port + 1 <= 65535 && port + 1 > 0) {
    return (port += 1);
  } else {
    throw new Error(
      "Invalid port value must be between 0-65535, got " + port + 1
    );
  }
};

module.exports = {
  changePort
};
